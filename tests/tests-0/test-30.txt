	static int Rfib(int n)
	{
		int first;
		int second;

			return n;
		}
		else
		{
			first = n - 1;
			first = Rfib(first);
			second = n - 2;
			second = Rfib(second);
			return first + second;
		}
	}

	static int Ifib(int n)
	{
		int first = 0;
		int second = 1;

		while(i < n)
